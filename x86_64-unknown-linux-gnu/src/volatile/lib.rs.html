1//! Provides the wrapper type `Volatile`, which wraps a reference to any copy-able type and allows
2//! for volatile memory access to wrapped value. Volatile memory accesses are never optimized away
3//! by the compiler, and are useful in many low-level systems programming and concurrent contexts.
4//!
5//! The wrapper types *do not* enforce any atomicity guarantees; to also get atomicity, consider
6//! looking at the `Atomic` wrapper types found in `libcore` or `libstd`.
7
8#![no_std]
9#![cfg_attr(feature = "unstable", feature(core_intrinsics))]
10#![cfg_attr(feature = "unstable", feature(slice_range))]
11#![cfg_attr(feature = "unstable", allow(incomplete_features))]
12#![cfg_attr(all(feature = "unstable", test), feature(slice_as_chunks))]

33/// Wraps a reference to make accesses to the referenced value volatile.
34///
35/// Allows volatile reads and writes on the referenced value. The referenced value needs to
36/// be `Copy` for reading and writing, as volatile reads and writes take and return copies
37/// of the value.
38///
39/// Since not all volatile resources (e.g. memory mapped device registers) are both readable
40/// and writable, this type supports limiting the allowed access types through an optional second
41/// generic parameter `A` that can be one of `ReadWrite`, `ReadOnly`, or `WriteOnly`. It defaults
42/// to `ReadWrite`, which allows all operations.
43///
44/// The size of this struct is the same as the size of the contained reference.
45#[derive(Clone)]
46#[repr(transparent)]
47pub struct Volatile<R, A = ReadWrite> {
48    reference: R,
49    access: PhantomData<A>,
50}

52/// Constructor functions for creating new values
53///
54/// These functions allow to construct a new `Volatile` instance from a reference type. While
55/// the `new` function creates a `Volatile` instance with unrestricted access, there are also
56/// functions for creating read-only or write-only instances.
57impl<R> Volatile<R> {
58    /// Constructs a new volatile instance wrapping the given reference.
59    ///
60    /// While it is possible to construct `Volatile` instances from arbitrary values (including
61    /// non-reference values), most of the methods are only available when the wrapped type is
62    /// a reference. The only reason that we don't forbid non-reference types in the constructor
63    /// functions is that the Rust compiler does not support trait bounds on generic `const`
64    /// functions yet. When this becomes possible, we will release a new version of this library
65    /// with removed support for non-references. For these reasons it is recommended to use
66    /// the `Volatile` type only with references.
67    ///
68    /// ## Example
69    ///
70    /// ```rust
71    /// use volatile::Volatile;
72    ///
73    /// let mut value = 0u32;
74    ///
75    /// let mut volatile = Volatile::new(&mut value);
76    /// volatile.write(1);
77    /// assert_eq!(volatile.read(), 1);
78    /// ```
79    pub const fn new(reference: R) -> Volatile<R> {
80        Volatile {
81            reference,
82            access: PhantomData,
83        }
84    }
85
86    /// Constructs a new read-only volatile instance wrapping the given reference.
87    ///
88    /// This is equivalent to the `new` function with the difference that the returned
89    /// `Volatile` instance does not permit write operations. This is for example useful
90    /// with memory-mapped hardware registers that are defined as read-only by the hardware.
91    ///
92    /// ## Example
93    ///
94    /// Reading is allowed:
95    ///
96    /// ```rust
97    /// use volatile::Volatile;
98    ///
99    /// let value = 0u32;
100    ///
101    /// let volatile = Volatile::new_read_only(&value);
102    /// assert_eq!(volatile.read(), 0);
103    /// ```
104    ///
105    /// But writing is not:
106    ///
107    /// ```compile_fail
108    /// use volatile::Volatile;
109    ///
110    /// let mut value = 0u32;
111    ///
112    /// let mut volatile = Volatile::new_read_only(&mut value);
113    /// volatile.write(1);
114    /// //ERROR: ^^^^^ the trait `volatile::access::Writable` is not implemented
115    /// //             for `volatile::access::ReadOnly`
116    /// ```
117    pub const fn new_read_only(reference: R) -> Volatile<R, ReadOnly> {
118        Volatile {
119            reference,
120            access: PhantomData,
121        }
122    }
123
124    /// Constructs a new write-only volatile instance wrapping the given reference.
125    ///
126    /// This is equivalent to the `new` function with the difference that the returned
127    /// `Volatile` instance does not permit read operations. This is for example useful
128    /// with memory-mapped hardware registers that are defined as write-only by the hardware.
129    ///
130    /// ## Example
131    ///
132    /// Writing is allowed:
133    ///
134    /// ```rust
135    /// use volatile::Volatile;
136    ///
137    /// let mut value = 0u32;
138    ///
139    /// let mut volatile = Volatile::new_write_only(&mut value);
140    /// volatile.write(1);
141    /// ```
142    ///
143    /// But reading is not:
144    ///
145    /// ```compile_fail
146    /// use volatile::Volatile;
147    ///
148    /// let value = 0u32;
149    ///
150    /// let volatile = Volatile::new_write_only(&value);
151    /// volatile.read();
152    /// //ERROR: ^^^^ the trait `volatile::access::Readable` is not implemented
153    /// //            for `volatile::access::WriteOnly`
154    /// ```
155    pub const fn new_write_only(reference: R) -> Volatile<R, WriteOnly> {
156        Volatile {
157            reference,
158            access: PhantomData,
159        }
160    }
161}

163/// Methods for references to `Copy` types
164impl<R, T, A> Volatile<R, A>
165where
166    R: Deref<Target = T>,
167    T: Copy,
168{
169    /// Performs a volatile read of the contained value.
170    ///
171    /// Returns a copy of the read value. Volatile reads are guaranteed not to be optimized
172    /// away by the compiler, but by themselves do not have atomic ordering
173    /// guarantees. To also get atomicity, consider looking at the `Atomic` wrapper types of
174    /// the standard/`core` library.
175    ///
176    /// ## Examples
177    ///
178    /// ```rust
179    /// use volatile::Volatile;
180    ///
181    /// let value = 42;
182    /// let shared_reference = Volatile::new(&value);
183    /// assert_eq!(shared_reference.read(), 42);
184    ///
185    /// let mut value = 50;
186    /// let mut_reference = Volatile::new(&mut value);
187    /// assert_eq!(mut_reference.read(), 50);
188    /// ```
189    pub fn read(&self) -> T
190    where
191        A: Readable,
192    {
193        // UNSAFE: Safe, as we know that our internal value exists.
194        unsafe { ptr::read_volatile(&*self.reference) }
195    }
196
197    /// Performs a volatile write, setting the contained value to the given `value`.
198    ///
199    /// Volatile writes are guaranteed to not be optimized away by the compiler, but by
200    /// themselves do not have atomic ordering guarantees. To also get atomicity, consider
201    /// looking at the `Atomic` wrapper types of the standard/`core` library.
202    ///
203    /// ## Example
204    ///
205    /// ```rust
206    /// use volatile::Volatile;
207    ///
208    /// let mut value = 42;
209    /// let mut volatile = Volatile::new(&mut value);
210    /// volatile.write(50);
211    ///
212    /// assert_eq!(volatile.read(), 50);
213    /// ```
214    pub fn write(&mut self, value: T)
215    where
216        A: Writable,
217        R: DerefMut,
218    {
219        // UNSAFE: Safe, as we know that our internal value exists.
220        unsafe { ptr::write_volatile(&mut *self.reference, value) };
221    }
222
223    /// Updates the contained value using the given closure and volatile instructions.
224    ///
225    /// Performs a volatile read of the contained value, passes a mutable reference to it to the
226    /// function `f`, and then performs a volatile write of the (potentially updated) value back to
227    /// the contained value.
228    ///
229    /// ```rust
230    /// use volatile::Volatile;
231    ///
232    /// let mut value = 42;
233    /// let mut volatile = Volatile::new(&mut value);
234    /// volatile.update(|val| *val += 1);
235    ///
236    /// assert_eq!(volatile.read(), 43);
237    /// ```
238    pub fn update<F>(&mut self, f: F)
239    where
240        A: Readable + Writable,
241        R: DerefMut,
242        F: FnOnce(&mut T),
243    {
244        let mut value = self.read();
245        f(&mut value);
246        self.write(value);
247    }
248}

250/// Method for extracting the wrapped value.
251impl<R, A> Volatile<R, A> {
252    /// Extracts the inner value stored in the wrapper type.
253    ///
254    /// This method gives direct access to the wrapped reference and thus allows
255    /// non-volatile access again. This is seldom what you want since there is usually
256    /// a reason that a reference is wrapped in `Volatile`. However, in some cases it might
257    /// be required or useful to use the `read_volatile`/`write_volatile` pointer methods of
258    /// the standard library directly, which this method makes possible.
259    ///
260    /// Since no memory safety violation can occur when accessing the referenced value using
261    /// non-volatile operations, this method is safe. However, it _can_ lead to bugs at the
262    /// application level, so this method should be used with care.
263    ///
264    /// ## Example
265    ///
266    /// ```
267    /// use volatile::Volatile;
268    ///
269    /// let mut value = 42;
270    /// let mut volatile = Volatile::new(&mut value);
271    /// volatile.write(50);
272    /// let unwrapped: &mut i32 = volatile.extract_inner();
273    ///
274    /// assert_eq!(*unwrapped, 50); // non volatile access, be careful!
275    /// ```
276    pub fn extract_inner(self) -> R {
277        self.reference
278    }
279}

281/// Transformation methods for accessing struct fields
282impl<R, T, A> Volatile<R, A>
283where
284    R: Deref<Target = T>,
285    T: ?Sized,
286{
287    /// Constructs a new `Volatile` reference by mapping the wrapped value.
288    ///
289    /// This method is useful for accessing individual fields of volatile structs.
290    ///
291    /// Note that this method gives temporary access to the wrapped reference, which allows
292    /// accessing the value in a non-volatile way. This is normally not what you want, so
293    /// **this method should only be used for reference-to-reference transformations**.
294    ///
295    /// ## Examples
296    ///
297    /// Accessing a struct field:
298    ///
299    /// ```
300    /// use volatile::Volatile;
301    ///
302    /// struct Example { field_1: u32, field_2: u8, }
303    /// let mut value = Example { field_1: 15, field_2: 255 };
304    /// let mut volatile = Volatile::new(&mut value);
305    ///
306    /// // construct a volatile reference to a field
307    /// let field_2 = volatile.map(|example| &example.field_2);
308    /// assert_eq!(field_2.read(), 255);
309    /// ```
310    ///
311    /// Don't misuse this method to do a non-volatile read of the referenced value:
312    ///
313    /// ```
314    /// use volatile::Volatile;
315    ///
316    /// let mut value = 5;
317    /// let mut volatile = Volatile::new(&mut value);
318    ///
319    /// // DON'T DO THIS:
320    /// let mut readout = 0;
321    /// volatile.map(|value| {
322    ///    readout = *value; // non-volatile read, might lead to bugs
323    ///    value
324    /// });
325    /// ```
326    pub fn map<'a, F, U>(&'a self, f: F) -> Volatile<&'a U, A>
327    where
328        F: FnOnce(&'a T) -> &'a U,
329        U: ?Sized,
330        T: 'a,
331    {
332        Volatile {
333            reference: f(self.reference.deref()),
334            access: self.access,
335        }
336    }
337
338    /// Constructs a new mutable `Volatile` reference by mapping the wrapped value.
339    ///
340    /// This method is useful for accessing individual fields of volatile structs.
341    ///
342    /// Note that this method gives temporary access to the wrapped reference, which allows
343    /// accessing the value in a non-volatile way. This is normally not what you want, so
344    /// **this method should only be used for reference-to-reference transformations**.
345    ///
346    /// ## Examples
347    ///
348    /// Accessing a struct field:
349    ///
350    /// ```
351    /// use volatile::Volatile;
352    ///
353    /// struct Example { field_1: u32, field_2: u8, }
354    /// let mut value = Example { field_1: 15, field_2: 255 };
355    /// let mut volatile = Volatile::new(&mut value);
356    ///
357    /// // construct a volatile reference to a field
358    /// let mut field_2 = volatile.map_mut(|example| &mut example.field_2);
359    /// field_2.write(128);
360    /// assert_eq!(field_2.read(), 128);
361    /// ```
362    ///
363    /// Don't misuse this method to do a non-volatile read or write of the referenced value:
364    ///
365    /// ```
366    /// use volatile::Volatile;
367    ///
368    /// let mut value = 5;
369    /// let mut volatile = Volatile::new(&mut value);
370    ///
371    /// // DON'T DO THIS:
372    /// volatile.map_mut(|value| {
373    ///    *value = 10; // non-volatile write, might lead to bugs
374    ///    value
375    /// });
376    /// ```
377    pub fn map_mut<'a, F, U>(&'a mut self, f: F) -> Volatile<&'a mut U, A>
378    where
379        F: FnOnce(&mut T) -> &mut U,
380        R: DerefMut,
381        U: ?Sized,
382        T: 'a,
383    {
384        Volatile {
385            reference: f(&mut self.reference),
386            access: self.access,
387        }
388    }
389}

391/// Methods for volatile slices
392impl<T, R, A> Volatile<R, A>
393where
394    R: Deref<Target = [T]>,
395{
396    /// Applies the index operation on the wrapped slice.
397    ///
398    /// Returns a shared `Volatile` reference to the resulting subslice.
399    ///
400    /// This is a convenience method for the `map(|slice| slice.index(index))` operation, so it
401    /// has the same behavior as the indexing operation on slice (e.g. panic if index is
402    /// out-of-bounds).
403    ///
404    /// ## Examples
405    ///
406    /// Accessing a single slice element:
407    ///
408    /// ```
409    /// use volatile::Volatile;
410    ///
411    /// let array = [1, 2, 3];
412    /// let slice = &array[..];
413    /// let volatile = Volatile::new(slice);
414    /// assert_eq!(volatile.index(1).read(), 2);
415    /// ```
416    ///
417    /// Accessing a subslice:
418    ///
419    /// ```
420    /// use volatile::Volatile;
421    ///
422    /// let array = [1, 2, 3];
423    /// let slice = &array[..];
424    /// let volatile = Volatile::new(slice);
425    /// let subslice = volatile.index(1..);
426    /// assert_eq!(subslice.index(0).read(), 2);
427    /// ```
428    pub fn index<'a, I>(&'a self, index: I) -> Volatile<&'a I::Output, A>
429    where
430        I: SliceIndex<[T]>,
431        T: 'a,
432    {
433        self.map(|slice| slice.index(index))
434    }
435
436    /// Applies the mutable index operation on the wrapped slice.
437    ///
438    /// Returns a mutable `Volatile` reference to the resulting subslice.
439    ///
440    /// This is a convenience method for the `map_mut(|slice| slice.index_mut(index))`
441    /// operation, so it has the same behavior as the indexing operation on slice
442    /// (e.g. panic if index is out-of-bounds).
443    ///
444    /// ## Examples
445    ///
446    /// Accessing a single slice element:
447    ///
448    /// ```
449    /// use volatile::Volatile;
450    ///
451    /// let mut array = [1, 2, 3];
452    /// let slice = &mut array[..];
453    /// let mut volatile = Volatile::new(slice);
454    /// volatile.index_mut(1).write(6);
455    /// assert_eq!(volatile.index(1).read(), 6);
456    /// ```
457    ///
458    /// Accessing a subslice:
459    ///
460    /// ```
461    /// use volatile::Volatile;
462    ///
463    /// let mut array = [1, 2, 3];
464    /// let slice = &mut array[..];
465    /// let mut volatile = Volatile::new(slice);
466    /// let mut subslice = volatile.index_mut(1..);
467    /// subslice.index_mut(0).write(6);
468    /// assert_eq!(subslice.index(0).read(), 6);
469    /// ```
470    pub fn index_mut<'a, I>(&'a mut self, index: I) -> Volatile<&mut I::Output, A>
471    where
472        I: SliceIndex<[T]>,
473        R: DerefMut,
474        T: 'a,
475    {
476        self.map_mut(|slice| slice.index_mut(index))
477    }

479    /// Copies all elements from `self` into `dst`, using a volatile memcpy.
480    ///
481    /// The length of `dst` must be the same as `self`.
482    ///
483    /// The method is only available with the `unstable` feature enabled (requires a nightly
484    /// Rust compiler).
485    ///
486    /// ## Panics
487    ///
488    /// This function will panic if the two slices have different lengths.
489    ///
490    /// ## Examples
491    ///
492    /// Copying two elements from a volatile slice:
493    ///
494    /// ```
495    /// use volatile::Volatile;
496    ///
497    /// let src = [1, 2];
498    /// // the `Volatile` type does not work with arrays, so convert `src` to a slice
499    /// let slice = &src[..];
500    /// let volatile = Volatile::new(slice);
501    /// let mut dst = [5, 0, 0];
502    ///
503    /// // Because the slices have to be the same length,
504    /// // we slice the destination slice from three elements
505    /// // to two. It will panic if we don't do this.
506    /// volatile.copy_into_slice(&mut dst[1..]);
507    ///
508    /// assert_eq!(src, [1, 2]);
509    /// assert_eq!(dst, [5, 1, 2]);
510    /// ```
511    #[cfg(feature = "unstable")]
512    pub fn copy_into_slice(&self, dst: &mut [T])
513    where
514        T: Copy,
515    {
516        let src = self.reference.deref();
517        assert_eq!(
518            src.len(),
519            dst.len(),
520            "destination and source slices have different lengths"
521        );
522        unsafe {
523            intrinsics::volatile_copy_nonoverlapping_memory(
524                dst.as_mut_ptr(),
525                src.as_ptr(),
526                src.len(),
527            );
528        }
529    }
530
531    /// Copies all elements from `src` into `self`, using a volatile memcpy.
532    ///
533    /// The length of `src` must be the same as `self`.
534    ///
535    /// This method is similar to the `slice::copy_from_slice` method of the standard library. The
536    /// difference is that this method performs a volatile copy.
537    ///
538    /// The method is only available with the `unstable` feature enabled (requires a nightly
539    /// Rust compiler).
540    ///
541    /// ## Panics
542    ///
543    /// This function will panic if the two slices have different lengths.
544    ///
545    /// ## Examples
546    ///
547    /// Copying two elements from a slice into a volatile slice:
548    ///
549    /// ```
550    /// use volatile::Volatile;
551    ///
552    /// let src = [1, 2, 3, 4];
553    /// let mut dst = [0, 0];
554    /// // the `Volatile` type does not work with arrays, so convert `dst` to a slice
555    /// let slice = &mut dst[..];
556    /// let mut volatile = Volatile::new(slice);
557    ///
558    /// // Because the slices have to be the same length,
559    /// // we slice the source slice from four elements
560    /// // to two. It will panic if we don't do this.
561    /// volatile.copy_from_slice(&src[2..]);
562    ///
563    /// assert_eq!(src, [1, 2, 3, 4]);
564    /// assert_eq!(dst, [3, 4]);
565    /// ```
566    #[cfg(feature = "unstable")]
567    pub fn copy_from_slice(&mut self, src: &[T])
568    where
569        T: Copy,
570        R: DerefMut,
571    {
572        let dest = self.reference.deref_mut();
573        assert_eq!(
574            dest.len(),
575            src.len(),
576            "destination and source slices have different lengths"
577        );
578        unsafe {
579            intrinsics::volatile_copy_nonoverlapping_memory(
580                dest.as_mut_ptr(),
581                src.as_ptr(),
582                dest.len(),
583            );
584        }
585    }
586
587    /// Copies elements from one part of the slice to another part of itself, using a
588    /// volatile `memmove`.
589    ///
590    /// `src` is the range within `self` to copy from. `dest` is the starting index of the
591    /// range within `self` to copy to, which will have the same length as `src`. The two ranges
592    /// may overlap. The ends of the two ranges must be less than or equal to `self.len()`.
593    ///
594    /// This method is similar to the `slice::copy_within` method of the standard library. The
595    /// difference is that this method performs a volatile copy.
596    ///
597    /// This method is only available with the `unstable` feature enabled (requires a nightly
598    /// Rust compiler).
599    ///
600    /// ## Panics
601    ///
602    /// This function will panic if either range exceeds the end of the slice, or if the end
603    /// of `src` is before the start.
604    ///
605    /// ## Examples
606    ///
607    /// Copying four bytes within a slice:
608    ///
609    /// ```
610    /// use volatile::Volatile;
611    ///
612    /// let mut byte_array = *b"Hello, World!";
613    /// let mut slice: &mut [u8] = &mut byte_array[..];
614    /// let mut volatile = Volatile::new(slice);
615    ///
616    /// volatile.copy_within(1..5, 8);
617    ///
618    /// assert_eq!(&byte_array, b"Hello, Wello!");
619    #[cfg(feature = "unstable")]
620    pub fn copy_within(&mut self, src: impl RangeBounds<usize>, dest: usize)
621    where
622        T: Copy,
623        R: DerefMut,
624    {
625        let slice = self.reference.deref_mut();
626        // implementation taken from https://github.com/rust-lang/rust/blob/683d1bcd405727fcc9209f64845bd3b9104878b8/library/core/src/slice/mod.rs#L2726-L2738
627        let Range {
628            start: src_start,
629            end: src_end,
630        } = range(src, ..slice.len());
631        let count = src_end - src_start;
632        assert!(dest <= slice.len() - count, "dest is out of bounds");
633        // SAFETY: the conditions for `volatile_copy_memory` have all been checked above,
634        // as have those for `ptr::add`.
635        unsafe {
636            intrinsics::volatile_copy_memory(
637                slice.as_mut_ptr().add(dest),
638                slice.as_ptr().add(src_start),
639                count,
640            );
641        }
642    }
643}

645/// Methods for volatile byte slices
646impl<R, A> Volatile<R, A>
647where
648    R: Deref<Target = [u8]>,
649{
650    /// Sets all elements of the byte slice to the given `value` using a volatile `memset`.
651    ///
652    /// This method is similar to the `slice::fill` method of the standard library, with the
653    /// difference that this method performs a volatile write operation. Another difference
654    /// is that this method is only available for byte slices (not general `&mut [T]` slices)
655    /// because there currently isn't a instrinsic function that allows non-`u8` values.
656    ///
657    /// This method is only available with the `unstable` feature enabled (requires a nightly
658    /// Rust compiler).
659    ///
660    /// ## Example
661    ///
662    /// ```rust
663    /// use volatile::Volatile;
664    ///
665    /// let mut buf = Volatile::new(vec![0; 10]);
666    /// buf.fill(1);
667    /// assert_eq!(buf.extract_inner(), vec![1; 10]);
668    /// ```
669    #[cfg(feature = "unstable")]
670    pub fn fill(&mut self, value: u8)
671    where
672        R: DerefMut,
673    {
674        let dest = self.reference.deref_mut();
675        unsafe {
676            intrinsics::volatile_set_memory(dest.as_mut_ptr(), value, dest.len());
677        }
678    }
679}

681/// Methods for converting arrays to slices
682impl<R, A, T, const N: usize> Volatile<R, A>
683where
684    R: Deref<Target = [T; N]>,
685{
686    /// Converts an array reference to a shared slice.
687    ///
688    /// This makes it possible to use the methods defined on slices.
689    ///
690    /// ## Example
691    ///
692    /// Reading a subslice from a volatile array reference using `index`:
693    ///
694    /// ```
695    /// use volatile::Volatile;
696    ///
697    /// let src = [1, 2, 3, 4];
698    /// let volatile = Volatile::new(&src);
699    ///
700    /// // convert the `Volatile<&[i32; 4]>` array reference to a `Volatile<&[i32]>` slice
701    /// let volatile_slice = volatile.as_slice();
702    /// // we can now use the slice methods
703    /// let subslice = volatile_slice.index(2..);
704    ///
705    /// assert_eq!(subslice.index(0).read(), 3);
706    /// assert_eq!(subslice.index(1).read(), 4);
707    /// ```
708    pub fn as_slice(&self) -> Volatile<&[T], A> {
709        self.map(|array| &array[..])
710    }
711
712    /// Converts a mutable array reference to a mutable slice.
713    ///
714    /// This makes it possible to use the methods defined on slices.
715    ///
716    /// ## Example
717    ///
718    /// Writing to an index of a mutable array reference:
719    ///
720    /// ```
721    /// use volatile::Volatile;
722    ///
723    /// let mut dst = [0, 0];
724    /// let mut volatile = Volatile::new(&mut dst);
725    ///
726    /// // convert the `Volatile<&mut [i32; 2]>` array reference to a `Volatile<&mut [i32]>` slice
727    /// let mut volatile_slice = volatile.as_mut_slice();
728    /// // we can now use the slice methods
729    /// volatile_slice.index_mut(1).write(1);
730    ///
731    /// assert_eq!(dst, [0, 1]);
732    /// ```
733    pub fn as_mut_slice(&mut self) -> Volatile<&mut [T], A>
734    where
735        R: DerefMut,
736    {
737        self.map_mut(|array| &mut array[..])
738    }
739}

741/// Methods for restricting access.
742impl<R> Volatile<R> {
743    /// Restricts access permissions to read-only.
744    ///
745    /// ## Example
746    ///
747    /// ```
748    /// use volatile::Volatile;
749    ///
750    /// let mut value: i16 = -4;
751    /// let mut volatile = Volatile::new(&mut value);
752    ///
753    /// let read_only = volatile.read_only();
754    /// assert_eq!(read_only.read(), -4);
755    /// // read_only.write(10); // compile-time error
756    /// ```
757    pub fn read_only(self) -> Volatile<R, ReadOnly> {
758        Volatile {
759            reference: self.reference,
760            access: PhantomData,
761        }
762    }
763
764    /// Restricts access permissions to write-only.
765    ///
766    /// ## Example
767    ///
768    /// Creating a write-only reference to a struct field:
769    ///
770    /// ```
771    /// use volatile::Volatile;
772    ///
773    /// struct Example { field_1: u32, field_2: u8, }
774    /// let mut value = Example { field_1: 15, field_2: 255 };
775    /// let mut volatile = Volatile::new(&mut value);
776    ///
777    /// // construct a volatile write-only reference to `field_2`
778    /// let mut field_2 = volatile.map_mut(|example| &mut example.field_2).write_only();
779    /// field_2.write(14);
780    /// // field_2.read(); // compile-time error
781    /// ```
782    pub fn write_only(self) -> Volatile<R, WriteOnly> {
783        Volatile {
784            reference: self.reference,
785            access: PhantomData,
786        }
787    }
788}
