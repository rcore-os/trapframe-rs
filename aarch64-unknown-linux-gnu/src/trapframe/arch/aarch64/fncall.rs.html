1//! Switch context by function call within the same privilege level.
2//!
3//! # Assumption
4//!
5//! This module suppose you are running kernel on Linux with glibc,
6//! and your user program is based on musl libc.
7//!
8//! Because we will store values in their pthread structure.
9
10use super::UserContext;
11use core::arch::global_asm;
12
13global_asm!(include_str!("fncall.S"));
14
15extern "C" {
16    /// The syscall entry of function call.
17    ///
18    /// # Usage
19    ///
20    /// Replace `svc` instruction by a `bl` instruction.
21    ///
22    /// ```asm
23    /// svc #0
24    /// bl syscall_fn_entry
25    /// ```
26    pub fn syscall_fn_entry();
27
28    fn syscall_fn_return(regs: &mut UserContext);
29}
30
31impl UserContext {
32    /// Go to user context by function return, within the same privilege level.
33    ///
34    /// User program should call `syscall_fn_entry()` to return back.
35    pub fn run_fncall(&mut self) {
36        unsafe {
37            syscall_fn_return(self);
38        }
39    }
40}
